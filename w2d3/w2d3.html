    const Employee = (function(){
        let name;
        let age;
        let salary;
        
        function getName(){
            return name;
        }
        function getAge(){
            return age;
        }
        function getSalary(){
            return salary;
        }
        function setName(n){
            name =  n;
        }
        function setAge(a){
            age =  a;
        }
        function setSalary(s){
            salary =  s;
        }
        function increaseSalary(percentage){
            setSalary(getSalary()+percentage/100*getSalary());
        }
        function incrementAge(){
            setAge(getAge()++);
        }
        
        return {
            setName : setName,
            setAge : setAge,
            setSalary : setSalary,
            increaseSalary : increaseSalary,
            incrementAge : incrementAge
        };        
    })();
